create table ydflix (
    num number primary key,
    id VARCHAR2(30),
    pw VARCHAR2(30),
    name VARCHAR2(30),
    phoneNb char(20),
    email VARCHAR2(200)
);

select nvl (max(num),0)+1 from ydflix;

select * from ydflix ORDER BY num;

insert into ydflix values ('1','1','1','1','1','1@email');

delete from ydflix where num >= 1;

drop table ydflix;
