create table ydflix (
    num number PRIMARY key,
    id VARCHAR2(30) ,
    pw VARCHAR2(30),
    name VARCHAR2(30),
    phoneNb char(20),
    email VARCHAR2(200)
);

select nvl (max(num),0)+1 from ydflix;

select * from ydflix ORDER BY num;

delete from ydflix where num >= 1;

drop table ydflix;
